================================================================================
                            API OVERVIEW
================================================================================

REGISTRY API:
  await wam2Registry.initialize()              // Fetch from online registry
  wam2Registry.getAll()                        // Get all 100+ plugins
  wam2Registry.search(query)                   // Search by text
  wam2Registry.getByCategory(category)         // Filter by category
  wam2Registry.getInstruments()                // Get instruments only
  wam2Registry.getEffects()                    // Get effects only
  wam2Registry.getPluginUrl(plugin)            // Get load URL
  wam2Registry.getGroupedByCategory()          // Get all categories

OUTPUT WAM2 API:
  const wam = new OutputWAM2(audioContext)     // Create instance
  const gui = await wam.createGui()            // Get GUI element
  await wam.destroyGui()                       // Cleanup GUI
  await wam.loadPlugin(descriptor)             // Load plugin
  wam.noteOn(noteNumber, velocity)             // Play note
  wam.noteOff(noteNumber)                      // Stop note
  wam.sendControlChange(cc, value, channel)    // Send CC message
  wam.getPluginInfo()                          // Get plugin metadata
